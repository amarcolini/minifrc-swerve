#define TCAADDR 0x70

void tcaselect(uint8_t i) {
  if (i > 7)
    return;

  Wire.beginTransmission(TCAADDR);
  Wire.write(1 << i);
  Wire.endTransmission();
}

class swervepod {
public:
  double p = 0.006, d = 0.0;
  double target = 0.0;
  double last_power = 0.0;
  double mult = 1.0;
  double last_rotation = 0.0;
  unsigned long last_time;
  double last_error = 0.0;
  NoU_Motor &rotation, &drive;

  swervepod(NoU_Motor &rotation, NoU_Motor &drive)
      : rotation(rotation), drive(drive) {
    last_time = micros();
  }

  void update(double angle, double power) {
    if (angle == NAN) {
      rotation.set(0.0);
      drive.set(0.0);
      return;
    }
    double dT = (micros() - last_time) / 1000000.0;
    double error = fmod(target - angle + 180.0, 360.0) - 180.0;
    if (error < -180.0) {
      error += 360.0;
    }
    mult = -1.0;
    if (error > 90.0)
      error -= 180.0;
    else if (error < -90.0)
      error += 180.0;
    else
      mult = 1.0;
    // double mult = 1.0;
    // PestoLink.printfTerminal("%.2f, %.2f", angle, error);
    if (abs(error) > 3.0) {
      double output = -1.0 * ((p * error) + (d * (error - last_error) / dT));
      if (output > 0.01) {
        output += 0.1;
      } else if (output < -0.01) {
        output += -0.1;
      }
      last_rotation += constrain(output - last_rotation, dT * -6.0, dT * 6.0);
      rotation.set(output);
    } else {
      rotation.set(0.0);
      last_rotation = 0.0;
    }
    double target_power = power * cos(error * PI / 180.0);
    last_power += constrain(mult * power - last_power, dT * -3.0, dT * 3.0);
    last_power = constrain(last_power, -0.5, 0.5);
    drive.set(last_power);
    last_error = error;
    last_time = micros();
  }

  void drive_direction(double power) {
    drive.set(power * mult);
    rotation.set(0.0);
  }

  void disable() {
    drive.set(0.0);
    rotation.set(0.0);
  }
};
